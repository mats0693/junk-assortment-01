<script language="VBScript">
Option Explicit

'アラート表示
Const SHOW_ALERT = True


'Const C_SakiPathRoot = "C:\MyWork\業務関連資料\" 'バックアップ先フォルダ、指定・固定 
Const C_SakiPathRoot = "C:\Temp\" 'バックアップ先フォルダ、指定・固定 

Call Main()
WScript.Quit() 'おわり

'----------------------------------------
'-- メイン・ルーチン
'----------------------------------------
Private Function Main()
	
	Dim strSourcePath		  '元ファイルフルパス

	Dim FSO 				  'FileSystemObject
	Dim Arg 				  '引数

	Set FSO = CreateObject("Scripting.FileSystemObject")

	'引数を確認
	If WScript.Arguments.Count = 0 Then
		'使用方法を表示
		WScript.Arguments.ShowUsage
	End If
	
	dim s1
	dim sAll
	dim sMoto
	dim sSaki
	dim spl
	dim iCnt
	
	iCnt = 0
	'ファイルの順次処理
	For Each Arg In WScript.Arguments
		strSourcePath = Arg
		'ファイルが存在したら実行処理を呼び出す
		If FSO.FileExists(strSourcePath) Then
			Call MainWork(strSourcePath)
			
			sMoto = strSourcePath
			sSaki = getSakiFullName(strSourcePath)
			''s1 = sMoto & "→" & sSaki
			s1 = sMoto
			sAll = sAll & spl & s1
			spl = vbcrlf
			
			iCnt = iCnt + 1
		End If
	Next
	
	If iCnt > 0 Then
		'MsgBox "[" & iCnt & "]ファイル" & " コピーしました。" & vbcrlf & sAll
		
		'エクスプローラでバックアップ先を開く
		call ExploreOpenFolder(getSakiPath(sMoto))
	End If
	
End Function

'----------------------------------------
'-- 1ファイル処理のメイン・ルーチン
'----------------------------------------
Private Function MainWork(strMotoFullName)
	'ファイルが処理されたらtrue、それ以外はfalseが返る
	
	'元ファイル
	Dim strMotoPath
	Dim strMotoName
	Dim strMotoNameBase
	Dim strMotoNameExt
	'戻り値
	Dim rtn
	
	'処理用オブジェクト
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")

	'引数がファイルであることの確認
	If Not FSO.FileExists(strMotoFullName) Then
		If SHOW_ALERT Then
			MsgBox "対象がファイルではありません。", vbQuestion + vbSystemModal
		End If
		MainWork = false: Exit Function
	End If

	'元ファイル名の取得・編集
	strMotoPath = FSO.GetParentFolderName(strMotoFullName) & "\"
	strMotoName = FSO.GetFileName(strMotoFullName)
	strMotoNameBase = FSO.GetBaseName(strMotoFullName)
	strMotoNameExt = FSO.GetExtensionName(strMotoFullName)

	'ファイルが存在しない
	If Not FSO.FileExists(strMotoFullName) Then
		If SHOW_ALERT Then
			MsgBox "[ERROR]処理対象ファイルが存在しません。" & strMotoName, vbSystemModal
		End If
		MainWork = false: Exit Function
	End If
	
	'コピー先フォルダ
	Dim strSakiPath
	strSakiPath = getSakiPath(strMotoFullName)
	''MsgBox strSakiPath
	call makeSakiPath(strSakiPath)
	
	'コピー先ファイル名
	Dim strSakiFullName
	strSakiFullName = getSakiFullName(strMotoFullName)

	''ファイルのコピー
	FSO.CopyFile strMotoFullName, strSakiFullName
	
	'MsgBox "指定フォルダにコピーしました。" & vbcrlf & strMotoFullName & vbcrlf & strSakiFullName
	
	MainWork = rtn: Exit Function
End Function

'----------------------------------------
'-- 出力先のファイルパスを返す
'----------------------------------------
Private Function getSakiPath(strMotoFullName)
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	Dim strSakiPath
	Dim strMotoPath
	strMotoPath = FSO.GetParentFolderName(strMotoFullName) & "\"
	''strSakiPath = strMotoPath & "old"
	strSakiPath = strMotoPath & "_old"

	getSakiPath = strSakiPath
	Exit Function

End Function

'----------------------------------------
'-- パスのドライブ文字列を返す
'----------------------------------------
Private Function getDriveLetter(strPath)
    Dim FSO
    Dim strDriveName
    Dim strDriveLetter
    Set FSO = CreateObject("Scripting.FileSystemObject")
    ''C:を返します
    strDriveName = FSO.GetDriveName(strPath)
    ''Cを返します
    If strDriveName <> "" Then
        strDriveLetter = Left(strDriveName,1)
    End If
    Set FSO = Nothing
    getDriveLetter = strDriveLetter
    Exit Function
End Function


'----------------------------------------
'-- 出力先のファイルパスを返す
'----------------------------------------
Private Function getSakiFullName(strMotoFullName)

	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	Dim strSakiFullName
	Dim strSakiPath
	Dim strSakiName
	Dim strSakiNameBase
	Dim strSakiNameExt

	Dim strDateTime
	Dim strDate
	Dim strTime
	Dim strMotoDateTime
	Dim strMotoDate
	Dim strWorkDateTime
	Dim dateMoto
	Dim dateWork
	dateMoto = FSO.GetFile(strMotoFullName).DateLastModified
	dateWork = Now()
	'元ファイルの更新時刻文字列 yyyymmdd_hhmmss
	strMotoDateTime = MyFormatDateTime(dateMoto)
	strMotoDate = MyFormatDate(dateMoto)
	'システム時刻文字列 yyyymmdd_hhmmss
	strWorkDateTime = MyFormatDateTime(dateWork)
	
	'先ファイル名の作成
	strSakiPath = getSakiPath(strMotoFullName) & "\" '

	strSakiNameBase = FSO.GetBaseName(strMotoFullName)		'元ファイルのベース名
	strSakiNameExt = FSO.GetExtensionName(strMotoFullName)	'元ファイルの拡張子
	

	strSakiName = strSakiNameBase & "_" & strMotoDate & "a" & "." & strSakiNameExt
	strSakiFullName = strSakiPath & strSakiName

	If Not FSO.FolderExists(strSakiPath) Then
		getSakiFullName = strSakiFullName
		Exit Function
	ElseIf Not FSO.FileExists(strSakiFullName) Then
		getSakiFullName = strSakiFullName
		Exit Function
	End If

	Const C_renban_work = "abcdefghijklmnopqrstuvwxyz"
	Dim i
	Dim sChar
	For i=2 To 26
		sChar = Mid(C_renban_work,i,1)
		strSakiName = strSakiNameBase & "_" & strMotoDate & sChar & "." & strSakiNameExt
		strSakiFullName = strSakiPath & strSakiName
		If Not FSO.FileExists(strSakiPath & strSakiName) Then
			getSakiFullName = strSakiFullName
			Exit Function
		End If
	Next

	sChar = ""
	strSakiName = strSakiNameBase & "_" & strMotoDateTime & "_at" & strWorkDateTime & sChar & "." & strSakiNameExt
	strSakiFullName = strSakiPath & strSakiName

	getSakiFullName = strSakiFullName
	Exit Function

End Function


'----------------------------------------
' 日付の固定フォーマット
' @param  workDate(日付型)
' @return yyyymmdd_hhmmss書式の日付文字列
'----------------------------------------
Function MyFormatDateTime(workDate)
	dim strMotoDateTime
	dim strDate
	dim strTime
	strDate = RIGHT(YEAR(workDate), 4) _
	        & RIGHT("0" & MONTH(workDate), 2) _
	        & RIGHT("0" & DAY(workDate), 2) 
	strTime = RIGHT("0" & HOUR(workDate), 2) _
	        & RIGHT("0" & MINUTE(workDate), 2) _
	        & RIGHT("0" & SECOND(workDate), 2)
	strMotoDateTime = strDate & "_" & strTime
	MyFormatDateTime = strMotoDateTime
End Function

'----------------------------------------
' 日付の固定フォーマット
' @param  workDate(日付型)
' @return yyyymmdd書式の日付文字列
'----------------------------------------
Function MyFormatDate(workDate)
	dim strMotoDateTime
	dim strDate
	dim strTime
	strDate = RIGHT(YEAR(workDate), 4) _
	        & RIGHT("0" & MONTH(workDate), 2) _
	        & RIGHT("0" & DAY(workDate), 2) 
	MyFormatDate = strDate
End Function

'----------------------------------------
' コピー先フォルダ作成処理
'
' @param コピー先フォルダ名
' @return none
'----------------------------------------
Sub makeSakiPath(ByVal strSakiPath)
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	'対象ディレクトリの存在確認
	If Not strSakiPath = "" Then
		IF Not FSO.FolderExists(strSakiPath) Then
			'なかったら作る
			MyCreateFolder(strSakiPath)'独自関数
		End If
	End If
	Set FSO = Nothing
End Sub

'----------------------------------------
'-- フォルダ作成
'----------------------------------------
Private Sub MyCreateFolder(ByVal Folder)
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	Dim ParentFolder
	'WScript.Echo Folder
	ParentFolder = FSO.GetParentFolderName(Folder)
	If ParentFolder <> "" Then If Not FSO.FolderExists(ParentFolder) Then MyCreateFolder ParentFolder
	If Not FSO.FolderExists(Folder) Then FSO.CreateFolder Folder
End Sub

'----------------------------------
'エクスプローラでフォルダを開く
'----------------------------------
Function ExploreOpenFolder(strPath)

	dim myFSO
	Set myFSO = CreateObject("Scripting.FileSystemObject")
	Dim objShell		' Shell オブジェクト
	Set objShell = WScript.CreateObject("Shell.Application")
	If Err.Number = 0 Then
		objShell.Explore strPath
		''WScript.Echo strPath & " を開きました。"
	Else
		WScript.Echo "エラー: " & Err.Description
	End If
	Set objShell = Nothing

End Function
