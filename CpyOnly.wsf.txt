Option Explicit

'アラート表示
Const SHOW_ALERT = True

Call Main()
WScript.Quit() 'おわり

'----------------------------------------
'-- メイン・ルーチン
'----------------------------------------
Private Function Main()
	
	Dim strSourcePath		  '元ファイルフルパス

	Dim FSO 				  'FileSystemObject
	Dim Arg 				  '引数

	Set FSO = CreateObject("Scripting.FileSystemObject")

	'引数を確認
	If WScript.Arguments.Count = 0 Then
		'使用方法を表示
		WScript.Arguments.ShowUsage
	End If

	'ファイルの順次処理
	For Each Arg In WScript.Arguments
		strSourcePath = Arg
		'ファイルまたはフォルダが存在したら実行処理を呼び出す
		If FSO.FileExists(strSourcePath) Or FSO.FolderExists(strSourcePath) Then
			Call MainWork(strSourcePath)
		End If
	Next
	
	''MsgBox "変換処理が完了しました。"

End Function

'----------------------------------------
'-- 1ファイル処理のメイン・ルーチン
'----------------------------------------

	'元ファイル
	Dim strMotoPath
	Dim strMotoName

	Dim strMotoNameExt
	'戻り値
	Dim rtn
	
	'処理用オブジェクト
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")

'	'引数がファイルであることの確認
'	If Not FSO.FileExists(strMotoFullName) Then
'		If SHOW_ALERT Then
'			MsgBox "対象がファイルではありません。", vbQuestion + vbSystemModal
'		End If
'		MainWork = false: Exit Function
'	End If

	'元ファイル名の取得・編集
	strMotoPath = FSO.GetParentFolderName(strMotoFullName) & "\"
	strMotoName = FSO.GetFileName(strMotoFullName)
	strMotoNameBase = FSO.GetBaseName(strMotoFullName)
	strMotoNameExt = FSO.GetExtensionName(strMotoFullName)

'	'処理対象ファイルでない
'	If LCase(strMotoNameExt) <> "sql" And _

	'対象ファイルの存在チェック
	'ファイルまたはフォルダが存在しない
	dim forFolderWork

		'Msgbox "no file"
		'フォルダは存在する場合、処理対象はフォルダ
		If FSO.FolderExists(strMotoFullName) Then

			If SHOW_ALERT Then

'========================================
'フォルダオブジェクトの取得
'========================================
Function getMyFolder(strFileFullName)
	
	Dim FSO
	Set FSO = CreateObject("Scripting.FileSystemObject")
	
	Dim fl 'Folder
	Set fl = FSO.GetFolder(strFileFullName) ' フォルダの取得
	
	Set getMyFolder = fl
	
End Function

'========================================
'ファイルオブジェクトの取得
'========================================
Function getMyFile(strFileFullName)

	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")

	Dim fl
	Set fl = FSO.GetFile(strFileFullName)

	Set getMyFile = fl

End Function

'元フォルダのフルパス、先フォルダ名（パスなし）
'対象の元パスの最下層フォルダ名を変更
Private Function myFolderCopy(strMotoFullName, strSakiFullName)

	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")

    ' コピー先フォルダが存在しないときは作成する
    If FSO.FolderExists(strSakiFullName) <> True Then
        FSO.CreateFolder(strSakiFullName)
    End If

    'フォルダコピー
    FSO.CopyFolder strMotoFullName, strSakiFullName, True
    If Err.Number = 0 Then
        ''WScript.Echo "データをコピーしました。"
    Else
        WScript.Echo "エラー: " & Err.Description
    End If
	
End Function

'----------------------------------------
'-- 出力先のファイルパスを返す
'----------------------------------------
Private Function getSakiFullName(strMotoFullName)
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	Dim strSakiFullName
	Dim strSakiPath
	Dim strSakiName
	Dim strSakiNameBase
	Dim strSakiNameExt

	Dim strDateTime
	Dim strDate
	Dim strTime
	Dim strMotoDateTime
	Dim strWorkDateTime
	Dim dateMoto
	Dim dateWork
	dateMoto = FSO.GetFile(strMotoFullName).DateLastModified
	dateWork = Now()
	'元ファイルの更新時刻文字列 yyyymmdd_hhmmss
	strMotoDateTime = MyFormatDateTime(dateMoto)
	'システム時刻文字列 yyyymmdd_hhmmss
	strWorkDateTime = MyFormatDateTime(dateWork)
	
	'先ファイル名の作成
	strSakiPath = FSO.GetParentFolderName(strMotoFullName) & "\" '元ファイルと同じパス
	strSakiNameBase = FSO.GetBaseName(strMotoFullName)		'元ファイルのベース名
	strSakiNameExt = FSO.GetExtensionName(strMotoFullName)	'元ファイルの拡張子
	strSakiName = strSakiNameBase & "_" & strMotoDateTime & "." & strSakiNameExt
	''strSakiName = strSakiNameBase & "_" & strMotoDateTime & "[at" & strWorkDateTime & "]" & "." & strSakiNameExt
	strSakiFullName = strSakiPath & strSakiName

	getSakiFullName = strSakiFullName
	Exit Function
End Function


'----------------------------------------
' 日付の固定フォーマット
' @param  workDate(日付型)
' @return yyyymmdd_hhmmss書式の日付文字列
'----------------------------------------
Function MyFormatDateTime(workDate)
	dim strMotoDateTime
	dim strDate
	dim strTime
	strDate = RIGHT(YEAR(workDate), 4) _
	        & RIGHT("0" & MONTH(workDate), 2) _
	        & RIGHT("0" & DAY(workDate), 2) 
	strTime = RIGHT("0" & HOUR(workDate), 2) _
	        & RIGHT("0" & MINUTE(workDate), 2) _
	        & RIGHT("0" & SECOND(workDate), 2)
	strMotoDateTime = strDate & "_" & strTime
	MyFormatDateTime = strMotoDateTime
End Function

'----------------------------------------
'-- フォルダ作成
'----------------------------------------
Private Sub CreateFolder(ByVal Folder)
	Dim FSO 'FileSystemObject
	Set FSO = CreateObject("Scripting.FileSystemObject")
	Dim ParentFolder
	'WScript.Echo Folder
	ParentFolder = FSO.GetParentFolderName(Folder)
	If ParentFolder <> "" Then If Not FSO.FolderExists(ParentFolder) Then CreateFolder ParentFolder
	If Not FSO.FolderExists(Folder) Then FSO.CreateFolder Folder
End Sub
